class Solution {
public:
    int addDigits(int num) {
        int cur = 0; // initial answer
        while(num != 0){
            cur += num%10; // add tail number
            if(cur >= 10){ // if cur > 0, remain tail and + 1
                cur = cur % 10 + 1;
            }
            num /= 10;
        }
        return cur;
    }
};
