/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        
        ListNode *tmp = head;
        
        while(tmp != NULL){
            if(tmp->val != 100001) tmp->val = 100001;
            else return true;
            tmp = tmp->next;
        }
        
        return false;
    }
};
